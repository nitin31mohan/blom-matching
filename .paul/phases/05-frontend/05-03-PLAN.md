<objective>
## Goal
Port the prototype's four core mechanics into the React/TypeScript codebase:
1. **Group attractors** — forceX/Y pulling nodes to group centers (replaces forceLink)
2. **Throw physics** — velocity tracking on drag → momentum on release
3. **Drag-to-reassign** — drop within 110px of a group center → node joins that group
4. **Live fit scoring** — JS cosine similarity recomputes node color on every reassignment

Plus: header (event selector + stats bar + fit legend), trait bars in the detail panel,
"Move to group" buttons in the panel for click-based reassignment.

## Purpose
This is the Tony Stark canvas. Nodes are physics objects — they cluster by group attraction,
can be thrown across the canvas, snap to the nearest group on drop, and immediately recolor
green/amber/red to show fit quality. The panel lets the operator inspect and manually
reassign any attendee. This is the core human-in-the-loop tool.

    ```ts
    export const TRAIT_KEYS = [
      'Social energy',
      'Openness',
      'Conscientiousness',
      'Agreeableness',
      'Eco values',
    ] as const
    export type TraitKey = (typeof TRAIT_KEYS)[number]

    // Cosine similarity between two trait vectors
    function cosineSim(a: number[], b: number[]): number {
      let dot = 0, magA = 0, magB = 0
      for (let i = 0; i < a.length; i++) {
        dot += a[i] * b[i]
        magA += a[i] * a[i]
        magB += b[i] * b[i]
      }
      return magA === 0 || magB === 0 ? 0 : dot / (Math.sqrt(magA) * Math.sqrt(magB))
    }

    // Average cosine similarity of node with all peers in the same group
    export function fitScore(
      nodeTraits: number[],
      groupId: string,
      allNodes: Array<{ pipeline_user_id: string; group_id: string; traits: number[] }>
    ): number {
      const peers = allNodes.filter(
        (n) => n.group_id === groupId && n.traits !== nodeTraits
      )
      if (peers.length === 0) return 1
      const sims = peers.map((p) => cosineSim(nodeTraits, p.traits))
      return sims.reduce((a, b) => a + b, 0) / sims.length
    }

    export function fitColor(score: number): string {
      if (score > 0.68) return '#22c55e'
      if (score > 0.42) return '#f59e0b'
      return '#ef4444'
    }

    export function fitLabel(score: number): string {
      if (score > 0.68) return 'Great fit'
      if (score > 0.42) return 'Okay fit'
      return 'Poor fit'
    }
    ```

    **handleDragStart (consolidates ALL drag + click + throw + reassign):**
    Move ALL logic from AttendeeNode into here. The window-event approach is kept
    (no d3.drag needed) but now also tracks velocity:
    ```ts
    const handleDragStart = useCallback((e: React.MouseEvent, node: SimNode) => {
      e.preventDefault()
      simRef.current?.alphaTarget(0.3).restart()

      let hasMoved = false
      let lastClientX = e.clientX, lastClientY = e.clientY
      let velX = 0, velY = 0
      const svgEl = svgRef.current!
      const getPos = (ev: MouseEvent) => {
        const r = svgEl.getBoundingClientRect()
        return { x: ev.clientX - r.left, y: ev.clientY - r.top }
      }

      const onMove = (ev: MouseEvent) => {
        hasMoved = true
        velX = ev.clientX - lastClientX
        velY = ev.clientY - lastClientY
        lastClientX = ev.clientX; lastClientY = ev.clientY
        const p = getPos(ev)
        node.fx = p.x; node.fy = p.y
      }

      const onUp = (ev: MouseEvent) => {
        window.removeEventListener('mousemove', onMove)
        window.removeEventListener('mouseup', onUp)

        if (!hasMoved) {
          // Click — open detail panel
          node.fx = null; node.fy = null
          simRef.current?.alphaTarget(0)
          onNodeClick?.(node)
          return
        }

        const pos = getPos(ev)

        // Check for reassignment: nearest group center within 110px
        let nearest = node.group_id
        let nearestDist = Infinity
        for (const gl of layout) {
          const [cx, cy] = groupCenter(gl.group_id)
          const dist = Math.hypot(pos.x - cx, pos.y - cy)
          if (dist < nearestDist) { nearestDist = dist; nearest = gl.group_id }
        }
        if (nearestDist < 110 && nearest !== node.group_id) {
          node.group_id = nearest
          onReassign?.(node.pipeline_user_id, nearest)
          // forceX/Y will pick up new group_id on next tick automatically
          simRef.current?.alpha(0.4).restart()
        }

        // Throw physics
        node.fx = null; node.fy = null
        const speed = Math.sqrt(velX * velX + velY * velY)
        if (speed > 3) {
          node.vx = velX * 5; node.vy = velY * 5
          simRef.current?.alpha(0.5).restart()
        } else {
          simRef.current?.alphaTarget(0)
        }
      }

      window.addEventListener('mousemove', onMove)
      window.addEventListener('mouseup', onUp)
    }, [layout, width, height, onNodeClick, onReassign])
    ```
